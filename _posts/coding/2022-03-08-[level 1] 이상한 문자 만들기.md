def solution(s):
    ans_li=s.split(" ")

    for  i in range(len(ans_li)):
        ans=list(ans_li[i])

        for j in range(len(ans)):
            if j%2==0:
                ans[j]=ans[j].upper()
            else:
                ans[j]=ans[j].lower()

        ans_li[i]="".join(ans)

    return " ".join(ans_li)
